cic <- function(q) {
  x1 <- F10_[which(F10 >= q)[1]]; x1
  x2 <- eF00(x1); x2
  x3 <- F01_[which(F01 >= x2)[1]]; x3

  F11_[which(F11 >= q)[1]] - x3
}
